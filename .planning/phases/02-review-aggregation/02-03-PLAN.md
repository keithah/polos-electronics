**Add JavaScript for read-more toggle** at bottom of file in script section:

```javascript
// Read more toggle
document.querySelectorAll('.read-more-btn').forEach(btn => {
    btn.addEventListener('click', function() {
        const container = this.parentElement;
        const excerpt = container.querySelector('.review-excerpt');
        const full = container.querySelector('.review-full');
        const expanded = this.getAttribute('aria-expanded') === 'true';

        if (excerpt && full) {
            excerpt.hidden = !expanded;
            full.hidden = expanded;
            this.textContent = expanded ? 'Read more' : 'Show less';
            this.setAttribute('aria-expanded', !expanded);
        }
    });
});
```
