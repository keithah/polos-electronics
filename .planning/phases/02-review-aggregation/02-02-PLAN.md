**Complete workflow implementation:**

```yaml
name: Fetch Reviews

on:
  schedule:
    - cron: '0 0 1 * *'  # Monthly: first day of month at midnight UTC
  workflow_dispatch:

permissions:
  contents: write

jobs:
  fetch-reviews:
    timeout-minutes: 15
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        with:
          token: ${{ secrets.GITHUB_TOKEN }}

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Install Playwright
        run: |
          npm install playwright
          npx playwright install chromium --with-deps

      - name: Extract platforms metadata
        run: |
          # CRITICAL: Preserve platforms metadata from existing reviews.json
          # This is needed by Plan 02-03 templates for profile URLs
          if [ -f data/reviews.json ]; then
            jq '.platforms // {}' data/reviews.json > /tmp/platforms.json
            echo "Preserved platforms metadata"
          else
            echo '{}' > /tmp/platforms.json
            echo "No existing platforms metadata - will use empty object"
          fi

      - name: Fetch Google Reviews
        env:
          GOOGLE_API_KEY: ${{ secrets.GOOGLE_API_KEY }}
          GOOGLE_PLACE_ID: ${{ secrets.GOOGLE_PLACE_ID }}
        run: |
          # Fetch from Places API
          RESPONSE=$(curl -sf "https://maps.googleapis.com/maps/api/place/details/json?place_id=${GOOGLE_PLACE_ID}&fields=name,rating,reviews,user_ratings_total&key=${GOOGLE_API_KEY}" || echo '{"result":{}}')

          # Extract platform-level stats for platforms object
          echo "$RESPONSE" | jq '{
            overallRating: .result.rating,
            totalReviews: .result.user_ratings_total
          }' > /tmp/google-stats.json
          echo "Google stats: rating=$(jq -r '.overallRating' /tmp/google-stats.json), total=$(jq -r '.totalReviews' /tmp/google-stats.json)"

          # Transform and filter to 4-5 star reviews only
          echo "$RESPONSE" | jq '[.result.reviews[]? | select(.rating >= 4) | {
            id: ("google-" + (.time | tostring)),
            platform: "google",
            author: .author_name,
            rating: .rating,
            text: .text,
            date: .relative_time_description,
            relativeTime: .relative_time_description,
            profilePhoto: .profile_photo_url
          }]' > /tmp/google-reviews.json

          echo "Fetched $(jq 'length' /tmp/google-reviews.json) Google reviews (4-5 stars)"

      - name: Fetch Yelp Reviews
        env:
          YELP_API_KEY: ${{ secrets.YELP_API_KEY }}
          YELP_BUSINESS_ID: ${{ secrets.YELP_BUSINESS_ID }}
        run: |
          # Skip if no Yelp API key configured
          if [ -z "$YELP_API_KEY" ]; then
            echo "YELP_API_KEY not configured - skipping Yelp fetch"
            echo "[]" > /tmp/yelp-reviews.json
            exit 0
          fi

          # Use YELP_BUSINESS_ID from secrets
          BUSINESS_ID="${YELP_BUSINESS_ID:-polos-electronics-battle-ground}"

          # Fetch from Yelp Fusion API (returns max 3 review excerpts)
          RESPONSE=$(curl -sf "https://api.yelp.com/v3/businesses/${BUSINESS_ID}/reviews" \
            -H "Authorization: Bearer ${YELP_API_KEY}" || echo '{"reviews":[]}')

          # Extract platform-level stats for platforms object
          # Yelp Fusion API returns rating and review_count in business endpoint
          BUSINESS_RESPONSE=$(curl -sf "https://api.yelp.com/v3/businesses/${BUSINESS_ID}" \
            -H "Authorization: Bearer ${YELP_API_KEY}" || echo '{}')
          echo "$BUSINESS_RESPONSE" | jq '{
            overallRating: .rating,
            totalReviews: .review_count
          }' > /tmp/yelp-stats.json
          echo "Yelp stats: rating=$(jq -r '.overallRating' /tmp/yelp-stats.json), total=$(jq -r '.totalReviews' /tmp/yelp-stats.json)"

          # Transform and filter to 4-5 star reviews only
          echo "$RESPONSE" | jq '[.reviews[]? | select(.rating >= 4) | {
            id: .id,
            platform: "yelp",
            author: .user.name,
            rating: .rating,
            text: .text,
            date: .time_created,
            relativeTime: .time_created,
            profilePhoto: .user.image_url
          }]' > /tmp/yelp-reviews.json

          echo "Fetched $(jq 'length' /tmp/yelp-reviews.json) Yelp reviews (4-5 stars)"

      - name: Scrape HomeAdvisor Reviews
        run: |
          cat > /tmp/scrape-homeadvisor.js << 'EOF'
          const { chromium } = require('playwright');

          (async () => {
            const browser = await chromium.launch({ headless: true });
            const page = await browser.newPage();
            const reviews = [];

            try {
              await page.goto('https://www.homeadvisor.com/rated.PolosElectronicsInc.7529229.html', {
                waitUntil: 'networkidle',
                timeout: 30000
              });

              // Wait for reviews to load
              await page.waitForSelector('[data-testid="review-card"], .review-card, .pro-review', {
                timeout: 10000
              }).catch(() => console.log('Review selector not found, page may have different structure'));

              // Extract reviews - adapt selectors based on actual page structure
              const reviewElements = await page.$$('[data-testid="review-card"], .review-card, .pro-review, .review-item');

              for (const el of reviewElements.slice(0, 10)) {
                try {
                  const author = await el.$eval('[class*="author"], [class*="name"], .reviewer-name', e => e.textContent?.trim()).catch(() => 'HomeAdvisor User');
                  const rating = await el.$eval('[class*="rating"], [class*="stars"]', e => {
                    const stars = e.querySelectorAll('[class*="filled"], .star-filled, [data-filled="true"]').length;
                    return stars || parseInt(e.getAttribute('data-rating') || e.textContent?.match(/(\d)/)?.[1] || '5');
                  }).catch(() => 5);
                  const text = await el.$eval('[class*="text"], [class*="content"], .review-text, .review-content', e => e.textContent?.trim()).catch(() => '');
                  const date = await el.$eval('[class*="date"], time', e => e.textContent?.trim() || e.getAttribute('datetime')).catch(() => new Date().toISOString().split('T')[0]);

                  if (text && rating >= 4) {
                    reviews.push({
                      id: 'homeadvisor-' + Date.now() + '-' + reviews.length,
                      platform: 'homeadvisor',
                      author,
                      rating: parseInt(rating),
                      text,
                      date,
                      relativeTime: date,
                      profilePhoto: null
                    });
                  }
                } catch (e) {
                  console.log('Error extracting review:', e.message);
                }
              }
            } catch (e) {
              console.error('HomeAdvisor scrape error:', e.message);
            }

            await browser.close();
            console.log(JSON.stringify(reviews));
          })();
          EOF

          node /tmp/scrape-homeadvisor.js > /tmp/homeadvisor-reviews.json 2>/tmp/homeadvisor-error.log || echo "[]" > /tmp/homeadvisor-reviews.json
          echo "Scraped $(jq 'length' /tmp/homeadvisor-reviews.json) HomeAdvisor reviews (4-5 stars)"

      - name: Scrape Nextdoor Reviews
        run: |
          cat > /tmp/scrape-nextdoor.js << 'EOF'
          const { chromium } = require('playwright');

          (async () => {
            const browser = await chromium.launch({ headless: true });
            const page = await browser.newPage();
            const reviews = [];

            try {
              await page.goto('https://nextdoor.com/pages/polos-electronics-battle-ground-wa/', {
                waitUntil: 'networkidle',
                timeout: 30000
              });

              // Wait for page to load - Nextdoor is SPA so may need extra time
              await page.waitForTimeout(3000);

              // Look for recommendation elements - Nextdoor uses recommendations not star ratings
              const reviewElements = await page.$$('[class*="recommendation"], [class*="review"], [class*="testimonial"]');

              for (const el of reviewElements.slice(0, 10)) {
                try {
                  const author = await el.$eval('[class*="author"], [class*="name"]', e => e.textContent?.trim()).catch(() => 'Neighbor');
                  const text = await el.$eval('[class*="text"], [class*="content"], [class*="body"]', e => e.textContent?.trim()).catch(() => '');
                  const date = await el.$eval('[class*="date"], time', e => e.textContent?.trim() || e.getAttribute('datetime')).catch(() => new Date().toISOString().split('T')[0]);

                  if (text) {
                    reviews.push({
                      id: 'nextdoor-' + Date.now() + '-' + reviews.length,
                      platform: 'nextdoor',
                      author,
                      rating: null,  // Nextdoor uses recommendations, not ratings
                      text,
                      date,
                      relativeTime: date,
                      profilePhoto: null
                    });
                  }
                } catch (e) {
                  console.log('Error extracting review:', e.message);
                }
              }
            } catch (e) {
              console.error('Nextdoor scrape error:', e.message);
            }

            await browser.close();
            console.log(JSON.stringify(reviews));
          })();
          EOF

          node /tmp/scrape-nextdoor.js > /tmp/nextdoor-reviews.json 2>/tmp/nextdoor-error.log || echo "[]" > /tmp/nextdoor-reviews.json
          echo "Scraped $(jq 'length' /tmp/nextdoor-reviews.json) Nextdoor recommendations"

      - name: Update platforms with stats
        run: |
          # Merge platform stats into platforms metadata
          # Google stats
          if [ -f /tmp/google-stats.json ]; then
            jq --slurpfile stats /tmp/google-stats.json '
              .google = (.google // {}) + $stats[0]
            ' /tmp/platforms.json > /tmp/platforms-updated.json
            mv /tmp/platforms-updated.json /tmp/platforms.json
          fi
          # Yelp stats
          if [ -f /tmp/yelp-stats.json ]; then
            jq --slurpfile stats /tmp/yelp-stats.json '
              .yelp = (.yelp // {}) + $stats[0]
            ' /tmp/platforms.json > /tmp/platforms-updated.json
            mv /tmp/platforms-updated.json /tmp/platforms.json
          fi
          echo "Updated platforms with stats:"
          jq '.' /tmp/platforms.json

      - name: Merge all reviews into data/reviews.json
        run: |
          # Merge all review sources, preserving platforms metadata from Plan 02-01 (now with stats)
          jq -s '{
            reviews: (.[0] + .[1] + .[2] + .[3]),
            platforms: .[4],
            lastUpdated: (now | strftime("%Y-%m-%dT%H:%M:%SZ"))
          }' /tmp/google-reviews.json /tmp/yelp-reviews.json /tmp/homeadvisor-reviews.json /tmp/nextdoor-reviews.json /tmp/platforms.json > data/reviews.json

          echo "Merged reviews:"
          echo "  - Google: $(jq 'length' /tmp/google-reviews.json)"
          echo "  - Yelp: $(jq 'length' /tmp/yelp-reviews.json)"
          echo "  - HomeAdvisor: $(jq 'length' /tmp/homeadvisor-reviews.json)"
          echo "  - Nextdoor: $(jq 'length' /tmp/nextdoor-reviews.json)"
          echo "  - Total: $(jq '.reviews | length' data/reviews.json)"
          echo "  - Platforms preserved: $(jq '.platforms | keys | length' data/reviews.json)"
          echo "Platform stats captured:"
          echo "  - Google: rating=$(jq -r '.platforms.google.overallRating // "N/A"' data/reviews.json), total=$(jq -r '.platforms.google.totalReviews // "N/A"' data/reviews.json)"
          echo "  - Yelp: rating=$(jq -r '.platforms.yelp.overallRating // "N/A"' data/reviews.json), total=$(jq -r '.platforms.yelp.totalReviews // "N/A"' data/reviews.json)"

      - name: Commit and push if changed
        run: |
          git config --global user.name 'GitHub Action'
          git config --global user.email 'action@github.com'
          git add data/reviews.json

          if git diff --staged --quiet; then
            echo "No changes to reviews"
          else
            git commit -m "Update reviews data [skip ci]"
            git push
            echo "Reviews updated and pushed"
          fi
```

**Key implementation details:**

1. **Playwright installation:** Installs Node.js 20 and Playwright with Chromium for web scraping.

2. **Platforms metadata preservation (CRITICAL):** The "Extract platforms metadata" step extracts the `platforms` object from existing reviews.json BEFORE any fetching. This is merged back at the end to preserve profile URLs needed by Plan 02-03 templates.

3. **Google Reviews:** Uses existing secrets, filters with `select(.rating >= 4)`, transforms to normalized schema.

4. **Yelp Reviews:** Uses `${{ secrets.YELP_BUSINESS_ID }}` instead of hardcoded ID. Skips gracefully if no key.

5. **HomeAdvisor Scraping:** Playwright script navigates to profile page, extracts reviews with adaptive selectors, filters to 4-5 stars.

6. **Nextdoor Scraping:** Playwright script navigates to business page, extracts recommendations (no star ratings on Nextdoor - uses null rating).

7. **Error handling:** Each API/scraper step has fallback to empty array `[]` to prevent workflow failure.

8. **Merge logic:** Uses `jq -s` to combine all review arrays and merge with preserved platforms metadata.
