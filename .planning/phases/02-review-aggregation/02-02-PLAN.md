**Schedule:** Change from daily (`0 0 * * *`) to monthly (`0 0 1 * *` - first day of month at midnight UTC)

**Complete workflow implementation:**

```yaml
name: Fetch Reviews

on:
  schedule:
    - cron: '0 0 1 * *'  # Monthly: first day of month at midnight UTC
  workflow_dispatch:

permissions:
  contents: write

jobs:
  fetch-reviews:
    timeout-minutes: 10
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        with:
          token: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract and preserve manual reviews
        run: |
          # CRITICAL: Extract manual reviews BEFORE any API fetches
          # These are homeadvisor and nextdoor reviews that have no API
          if [ -f data/reviews.json ]; then
            jq '[.reviews[]? | select(.platform == "homeadvisor" or .platform == "nextdoor")]' data/reviews.json > /tmp/manual-reviews.json
            echo "Preserved $(jq 'length' /tmp/manual-reviews.json) manual reviews (homeadvisor/nextdoor)"
          else
            echo "[]" > /tmp/manual-reviews.json
            echo "No existing reviews.json - starting fresh"
          fi

      - name: Fetch Google Reviews
        env:
          GOOGLE_API_KEY: ${{ secrets.GOOGLE_API_KEY }}
          GOOGLE_PLACE_ID: ${{ secrets.GOOGLE_PLACE_ID }}
        run: |
          # Fetch from Places API
          RESPONSE=$(curl -sf "https://maps.googleapis.com/maps/api/place/details/json?place_id=${GOOGLE_PLACE_ID}&fields=name,rating,reviews,user_ratings_total&key=${GOOGLE_API_KEY}" || echo '{"result":{}}')

          # Transform and filter to 4-5 star reviews only
          echo "$RESPONSE" | jq '[.result.reviews[]? | select(.rating >= 4) | {
            id: ("google-" + (.time | tostring)),
            platform: "google",
            author: .author_name,
            rating: .rating,
            text: .text,
            date: .relative_time_description,
            profilePhoto: .profile_photo_url
          }]' > /tmp/google-reviews.json

          echo "Fetched $(jq 'length' /tmp/google-reviews.json) Google reviews (4-5 stars)"

      - name: Fetch Yelp Reviews
        env:
          YELP_API_KEY: ${{ secrets.YELP_API_KEY }}
        run: |
          # Skip if no Yelp API key configured
          if [ -z "$YELP_API_KEY" ]; then
            echo "YELP_API_KEY not configured - skipping Yelp fetch"
            echo "[]" > /tmp/yelp-reviews.json
            exit 0
          fi

          # Fetch from Yelp Fusion API (returns max 3 review excerpts)
          RESPONSE=$(curl -sf "https://api.yelp.com/v3/businesses/polos-electronics-battle-ground/reviews" \
            -H "Authorization: Bearer ${YELP_API_KEY}" || echo '{"reviews":[]}')

          # Transform and filter to 4-5 star reviews only
          echo "$RESPONSE" | jq '[.reviews[]? | select(.rating >= 4) | {
            id: .id,
            platform: "yelp",
            author: .user.name,
            rating: .rating,
            text: .text,
            date: .time_created,
            profilePhoto: .user.image_url
          }]' > /tmp/yelp-reviews.json

          echo "Fetched $(jq 'length' /tmp/yelp-reviews.json) Yelp reviews (4-5 stars)"

      - name: Merge all reviews into data/reviews.json
        run: |
          # Merge: preserved manual reviews + fresh google + fresh yelp
          # Manual reviews (homeadvisor, nextdoor) are NEVER overwritten
          jq -s '{
            reviews: (.[0] + .[1] + .[2]),
            lastUpdated: (now | strftime("%Y-%m-%dT%H:%M:%SZ"))
          }' /tmp/manual-reviews.json /tmp/google-reviews.json /tmp/yelp-reviews.json > data/reviews.json

          echo "Merged reviews:"
          echo "  - Manual (homeadvisor/nextdoor): $(jq 'length' /tmp/manual-reviews.json)"
          echo "  - Google: $(jq 'length' /tmp/google-reviews.json)"
          echo "  - Yelp: $(jq 'length' /tmp/yelp-reviews.json)"
          echo "  - Total: $(jq '.reviews | length' data/reviews.json)"

      - name: Commit and push if changed
        run: |
          git config --global user.name 'GitHub Action'
          git config --global user.email 'action@github.com'
          git add data/reviews.json

          if git diff --staged --quiet; then
            echo "No changes to reviews"
          else
            git commit -m "Update reviews data [skip ci]"
            git push
            echo "Reviews updated and pushed"
          fi
```

**Key implementation details:**

1. **Manual review preservation (CRITICAL):** The "Extract and preserve manual reviews" step runs FIRST, extracting homeadvisor and nextdoor reviews to `/tmp/manual-reviews.json` before any API calls. These are merged back in at the end.

2. **Google Reviews:** Uses existing secrets, filters with `select(.rating >= 4)`, transforms to normalized schema.

3. **Yelp Reviews:** Checks if `YELP_API_KEY` is set, skips gracefully if not. Uses Fusion API endpoint.

4. **Error handling:** Each curl uses `-sf` (silent + fail) with `|| echo '{}'` fallback to prevent workflow failure on API errors.

5. **Merge logic:** Uses `jq -s` to slurp all three JSON arrays and combine them, preserving manual reviews.
