**Create layouts/partials/head/critical-css.html:**
```hugo
{{/* Inline critical CSS for above-fold content */}}
{{ $critical := resources.Get "css/critical.css" }}
{{ if hugo.IsProduction }}
  {{ $critical = $critical | resources.Minify }}
{{ end }}
<style>{{ $critical.Content | safeCSS }}</style>
```

**Create layouts/partials/head/deferred-css.html:**
```hugo
{{/* Defer non-critical CSS using preload/swap pattern */}}
{{ $deferred := resources.Get "css/deferred.css" | resources.Minify | resources.Fingerprint "sha256" }}
<link rel="preload"
      href="{{ $deferred.RelPermalink }}"
      as="style"
      onload="this.onload=null;this.rel='stylesheet'">
<noscript>
  <link rel="stylesheet" href="{{ $deferred.RelPermalink }}">
</noscript>
```

Note: Do NOT include .section-title::after pseudo-element styles in critical CSS (those are for section headers below the fold). Only include the base .section-title if needed for hero.
